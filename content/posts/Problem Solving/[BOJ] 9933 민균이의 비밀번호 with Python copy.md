from sys import stdin

n = int(stdin.readline())
data = []
for _ in range(n):
    data.append(list(stdin.readline().rstrip()))


def solve():
    for i in range(n):
        temp = data[i][:]
        temp.reverse()
        if temp in data:

            mid = data[i][length // 2]
            return length, mid
